27 bool ompl::app::RigidBodyGeometry::addRobotMesh(const std::string &robot)
28 {
29  assert(!robot.empty());
30  std::size_t p = importerRobot_.size();
31  importerRobot_.resize(p + 1);
32  importerRobot_[p].reset(new Assimp::Importer());
33 
34  const aiScene* robotScene = importerRobot_[p]->ReadFile(robot.c_str(),
35  aiProcess_Triangulate |
36  aiProcess_JoinIdenticalVertices |
37  aiProcess_SortByPType |
38  aiProcess_OptimizeGraph |
39  aiProcess_OptimizeMeshes);
40  if (robotScene)
41  {
42  if (!robotScene->HasMeshes())
43  {
44  OMPL_ERROR("There is no mesh specified in the indicated robot resource: %s", robot.c_str());
45  importerRobot_.resize(p);
46  }
47  }
48  else
49  {
50  OMPL_ERROR("Unable to load robot scene: %s", robot.c_str());
51  importerRobot_.resize(p);
52  }
53 
54  if (p < importerRobot_.size())
55  {
56  computeGeometrySpecification();
57  return true;
58  }
59  else
60  return false;
61 }

131 void ompl::app::RigidBodyGeometry::computeGeometrySpecification(void)
132 {
133  validitySvc_.reset();
134  geom_.obstacles.clear();
135  geom_.obstaclesShift.clear();
136  geom_.robot.clear();
137  geom_.robotShift.clear();
138 
139  for (unsigned int i = 0 ; i < importerEnv_.size() ; ++i)
140  geom_.obstacles.push_back(importerEnv_[i]->GetScene());
141 
142  for (unsigned int i = 0 ; i < importerRobot_.size() ; ++i)
143  {
144  geom_.robot.push_back(importerRobot_[i]->GetScene());
145  aiVector3D c = getRobotCenter(i);
146  if (mtype_ == Motion_2D)
147  c[2] = 0.0;
148  geom_.robotShift.push_back(c);
149  }
150 }

176 const ompl::base::StateValidityCheckerPtr& ompl::app::RigidBodyGeometry::allocStateValidityChecker(const base::SpaceInformationPtr &si, const GeometricStateExtractor &se, bool selfCollision)
177 {
178  if (validitySvc_)
179  return validitySvc_;
180 
181  GeometrySpecification geom = getGeometrySpecification();
182 
183  switch (ctype_)
184  {
185 #if OMPL_HAS_PQP
186  case PQP:
187  if (mtype_ == Motion_2D)
188  validitySvc_.reset (new PQPStateValidityChecker<Motion_2D>(si, geom, se, selfCollision));
189  else
190  validitySvc_.reset (new PQPStateValidityChecker<Motion_3D>(si, geom, se, selfCollision));
191  break;
192 #endif
193  case FCL:
194  if (mtype_ == Motion_2D)
195  validitySvc_.reset (new FCLStateValidityChecker<Motion_2D>(si, geom, se, selfCollision));
196  else
197  validitySvc_.reset (new FCLStateValidityChecker<Motion_3D>(si, geom, se, selfCollision));
198  break;
199 
200  default:
201  OMPL_ERROR("Unexpected collision checker type (%d) encountered", ctype_);
202  };
203 
204  return validitySvc_;
205 }
