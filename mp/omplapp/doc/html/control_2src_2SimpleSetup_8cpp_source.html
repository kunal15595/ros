41 ompl::base::PlannerPtr ompl::control::getDefaultPlanner(const base::GoalPtr &goal)
42 {
43  base::PlannerPtr planner;
44  if (!goal)
45  throw Exception("Unable to allocate default planner for unspecified goal definition");
46 
47  SpaceInformationPtr si = boost::static_pointer_cast<SpaceInformation, base::SpaceInformation>(goal->getSpaceInformation());
48  if (si->getStateSpace()->hasDefaultProjection())
49  planner = base::PlannerPtr(new KPIECE1(si));
50  else
51  planner = base::PlannerPtr(new RRT(si));
52 
53  return planner;
54 }

72 void ompl::control::SimpleSetup::setup()
73 {
74  if (!configured_ || !si_->isSetup() || !planner_->isSetup())
75  {
76  if (!si_->isSetup())
77  si_->setup();
78  if (!planner_)
79  {
80  if (pa_)
81  planner_ = pa_(si_);
82  if (!planner_)
83  {
84  OMPL_INFORM("No planner specified. Using default.");
85  planner_ = getDefaultPlanner(getGoal());
86  }
87  }
88  planner_->setProblemDefinition(pdef_);
89  if (!planner_->isSetup())
90  planner_->setup();
91 
92  params_.clear();
93  params_.include(si_->params());
94  params_.include(planner_->params());
95  configured_ = true;
96  }
97 }

107 // we provide a duplicate implementation here to allow the planner to choose how the time is turned into a planner termination condition
108 ompl::base::PlannerStatus ompl::control::SimpleSetup::solve(double time)
109 {
110  setup();
111  last_status_ = base::PlannerStatus::UNKNOWN;
112  time::point start = time::now();
113  last_status_ = planner_->solve(time);
114  planTime_ = time::seconds(time::now() - start);
115  if (last_status_)
116  OMPL_INFORM("Solution found in %f seconds", planTime_);
117  else
118  OMPL_INFORM("No solution found after %f seconds", planTime_);
119  return last_status_;
120 }
