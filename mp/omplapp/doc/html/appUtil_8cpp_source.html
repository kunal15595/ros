19 void ompl::app::InferProblemDefinitionBounds(const base::ProblemDefinitionPtr &pdef, const GeometricStateExtractor &se, double factor, double add,
20  unsigned int robotCount, const base::StateSpacePtr &space, MotionModel mtype)
21 {
22  // update the bounds based on start states, if needed
23  base::RealVectorBounds bounds = mtype == Motion_2D ? space->as<base::SE2StateSpace>()->getBounds() : space->as<base::SE3StateSpace>()->getBounds();
24 
25  std::vector<const base::State*> states;
26  pdef->getInputStates(states);
27 
28  double minX = std::numeric_limits<double>::infinity();
29  double minY = minX;
30  double minZ = minX;
31  double maxX = -minX;
32  double maxY = maxX;
33  double maxZ = maxX;
34  for (unsigned int i = 0 ; i < states.size() ; ++i)
35  {
36  for (unsigned int r = 0 ; r < robotCount ; ++r)
37  {
38  const base::State *s = se(states[i], r);
39  double x = mtype == Motion_2D ? s->as<base::SE2StateSpace::StateType>()->getX() : s->as<base::SE3StateSpace::StateType>()->getX();
40  double y = mtype == Motion_2D ? s->as<base::SE2StateSpace::StateType>()->getY() : s->as<base::SE3StateSpace::StateType>()->getY();
41  double z = mtype == Motion_2D ? 0.0 : s->as<base::SE3StateSpace::StateType>()->getZ();
42  if (minX > x) minX = x;
43  if (maxX < x) maxX = x;
44  if (minY > y) minY = y;
45  if (maxY < y) maxY = y;
46  if (minZ > z) minZ = z;
47  if (maxZ < z) maxZ = z;
48  }
49  }
50  double dx = (maxX - minX) * (factor - 1.0) + add;
51  double dy = (maxY - minY) * (factor - 1.0) + add;
52  double dz = (maxZ - minZ) * (factor - 1.0) + add;
53 
54  if (bounds.low[0] > minX - dx) bounds.low[0] = minX - dx;
55  if (bounds.low[1] > minY - dy) bounds.low[1] = minY - dy;
56 
57  if (bounds.high[0] < maxX + dx) bounds.high[0] = maxX + dx;
58  if (bounds.high[1] < maxY + dy) bounds.high[1] = maxY + dy;
59 
60  if (mtype == Motion_3D)
61  {
62  if (bounds.high[2] < maxZ + dz) bounds.high[2] = maxZ + dz;
63  if (bounds.low[2] > minZ - dz) bounds.low[2] = minZ - dz;
64 
65  space->as<base::SE3StateSpace>()->setBounds(bounds);
66  }
67  else
68  space->as<base::SE2StateSpace>()->setBounds(bounds);
69 }

71 void ompl::app::InferEnvironmentBounds(const base::StateSpacePtr &space, const RigidBodyGeometry &rbg)
72 {
73  MotionModel mtype = rbg.getMotionModel();
74 
75  base::RealVectorBounds bounds = mtype == Motion_2D ? space->as<base::SE2StateSpace>()->getBounds() : space->as<base::SE3StateSpace>()->getBounds();
76 
77  // if bounds are not valid
78  if (bounds.getVolume() < std::numeric_limits<double>::epsilon())
79  {
80  if (mtype == Motion_2D)
81  space->as<base::SE2StateSpace>()->setBounds(rbg.inferEnvironmentBounds());
82  else
83  space->as<base::SE3StateSpace>()->setBounds(rbg.inferEnvironmentBounds());
84  }
