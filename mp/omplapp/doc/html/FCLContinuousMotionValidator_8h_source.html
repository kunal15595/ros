52  virtual bool checkMotion(const ob::State *s1, const ob::State *s2) const
53  {
54  double unused;
55 
56  // assume motion starts in a valid configuration so s1 is valid
57  // Must check validity of s2 before performing collision check between s1 and s2
58  bool valid = si_->isValid(s2) && fclWrapper_->isValid (s1, s2, unused);
59 
60  // Increment valid/invalid motion counters
61  valid ? valid_++ : invalid_++;
62 
63  return valid;
64  }

69  virtual bool checkMotion(const ob::State *s1, const ob::State *s2, std::pair<ob::State*, double> &lastValid) const
70  {
71  bool valid = false;
72 
73  // if there is a collision, collisionTime will contain the time to collision,
74  // parameterized from [0,1), where s1 is 0 and s2 is 1.
75  double collisionTime;
76  valid = fclWrapper_->isValid (s1, s2, collisionTime);
77 
78  // Find the last valid state before collision...
79  // NOTE: This should probably be refactored so that the continuous checker
80  // returns the last valid time. Last valid transformation may also be
81  // possible, but that introduces a dependency on the geometry.
82  if (!valid)
83  {
84  ob::State *lastValidState;
85  if (lastValid.first)
86  lastValidState = lastValid.first;
87  else
88  lastValidState = si_->allocState ();
89 
90  collisionTime -= 0.01;
91  stateSpace_->interpolate (s1, s2, collisionTime, lastValidState);
92 
93  while (!si_->isValid (lastValidState) && collisionTime > 0)
94  {
95  collisionTime -= 0.01;
96  stateSpace_->interpolate (s1, s2, collisionTime, lastValidState);
97  }
98 
99  // ensure that collisionTime is greater than zero
100  if (collisionTime < 0.01)
101  {
102  collisionTime = 0.0;
103  si_->copyState (lastValidState, s1);
104  }
105 
106  lastValid.second = collisionTime;
107 
108  if (!lastValid.first)
109  si_->freeState (lastValidState);
110  }
111 
112  // Increment valid/invalid motion counters
113  valid ? valid_++ : invalid_++;
114 
115  return valid;
116  }

121  void defaultSettings(MotionModel mm)
122  {
123  stateSpace_ = si_->getStateSpace().get();
124  if (!stateSpace_)
125  throw Exception("No state space for motion validator");
126 
127  // Extract FCLWrapper from FCLStateValidityChecker.
128  switch (mm)
129  {
130  case app::Motion_2D:
131  const app::FCLStateValidityChecker<app::Motion_2D> *fcl_2d_state_checker;
132  fcl_2d_state_checker = dynamic_cast <const app::FCLStateValidityChecker<app::Motion_2D>* > (si_->getStateValidityChecker ().get ());
133 
134  if (!fcl_2d_state_checker)
135  {
136  // Be extra verbose in this fatal error
137  OMPL_ERROR("Unable to cast state validity checker to FCLStateValidityChecker.");
138  assert (fcl_2d_state_checker != 0);
139  }
140 
141  fclWrapper_ = fcl_2d_state_checker->getFCLWrapper ();
142  break;
143 
144  case app::Motion_3D:
145  const app::FCLStateValidityChecker<app::Motion_3D> *fcl_3d_state_checker;
146  fcl_3d_state_checker = dynamic_cast <const app::FCLStateValidityChecker<app::Motion_3D>* > (si_->getStateValidityChecker ().get ());
147 
148  if (!fcl_3d_state_checker)
149  {
150  // Be extra verbose in this fatal error
151  OMPL_ERROR("Unable to cast state validity checker to FCLStateValidityChecker.");
152  assert (fcl_3d_state_checker != 0);
153  }
154 
155  fclWrapper_ = fcl_3d_state_checker->getFCLWrapper ();
156  break;
157 
158  default:
159  OMPL_WARN("Unknown motion model specified: %u", mm);
160  break;
161  }
162 
163  if (!fclWrapper_)
164  {
165  // Be extra verbose in this fatal error
166  OMPL_ERROR("FCLWrapper object is not valid.");
167  assert (fclWrapper_ != 0);
168  }
169  }
