47 bool ompl::control::OpenDEStateValidityChecker::isValid(const base::State *state) const
48 {
49  const OpenDEStateSpace::StateType *s = state->as<OpenDEStateSpace::StateType>();
50 
51  // if we know the value of the validity flag for this state, we return it
52  if (s->collision & (1 << OpenDEStateSpace::STATE_VALIDITY_KNOWN_BIT))
53  return s->collision & (1 << OpenDEStateSpace::STATE_VALIDITY_VALUE_BIT);
54 
55  // if not, we compute it:
56  bool valid = false;
57 
58  if (!osm_->evaluateCollision(state))
59  valid = osm_->satisfiesBoundsExceptRotation(s);
60 
61  if (valid)
62  s->collision &= (1 << OpenDEStateSpace::STATE_VALIDITY_VALUE_BIT);
63 
64  // mark the fact we know the value of the validity bit
65  s->collision &= (1 << OpenDEStateSpace::STATE_VALIDITY_KNOWN_BIT);
66 
67  return valid;
68 }
