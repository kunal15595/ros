44 ompl::base::PlannerPtr ompl::geometric::getDefaultPlanner(const base::GoalPtr &goal)
45 {
46  base::PlannerPtr planner;
47  if (!goal)
48  throw Exception("Unable to allocate default planner for unspecified goal definition");
49 
50  // if we can sample the goal region, use a bi-directional planner
51  if (goal->hasType(base::GOAL_SAMPLEABLE_REGION))
52  {
53  // if we have a default projection
54  if (goal->getSpaceInformation()->getStateSpace()->hasDefaultProjection())
55  planner = base::PlannerPtr(new LBKPIECE1(goal->getSpaceInformation()));
56  else
57  planner = base::PlannerPtr(new RRTConnect(goal->getSpaceInformation()));
58  }
59  // other use a single-tree planner
60  else
61  {
62  // if we have a default projection
63  if (goal->getSpaceInformation()->getStateSpace()->hasDefaultProjection())
64  planner = base::PlannerPtr(new KPIECE1(goal->getSpaceInformation()));
65  else
66  planner = base::PlannerPtr(new RRT(goal->getSpaceInformation()));
67  }
68 
69  if (!planner)
70  throw Exception("Unable to allocate default planner");
71 
72  return planner;
73 }

93 void ompl::geometric::SimpleSetup::setup()
94 {
95  if (!configured_ || !si_->isSetup() || !planner_->isSetup())
96  {
97  if (!si_->isSetup())
98  si_->setup();
99  if (!planner_)
100  {
101  if (pa_)
102  planner_ = pa_(si_);
103  if (!planner_)
104  {
105  OMPL_INFORM("No planner specified. Using default.");
106  planner_ = getDefaultPlanner(getGoal());
107  }
108  }
109  planner_->setProblemDefinition(pdef_);
110  if (!planner_->isSetup())
111  planner_->setup();
112 
113  params_.clear();
114  params_.include(si_->params());
115  params_.include(planner_->params());
116  configured_ = true;
117  }
118 }

157 void ompl::geometric::SimpleSetup::simplifySolution(const base::PlannerTerminationCondition &ptc)
158 {
159  if (pdef_)
160  {
161  const base::PathPtr &p = pdef_->getSolutionPath();
162  if (p)
163  {
164  time::point start = time::now();
165  PathGeometric &path = static_cast<PathGeometric&>(*p);
166  std::size_t numStates = path.getStateCount();
167  psk_->simplify(path, ptc);
168  simplifyTime_ = time::seconds(time::now() - start);
169  OMPL_INFORM("Path simplification took %f seconds and changed from %d to %d states",
170  simplifyTime_, numStates, path.getStateCount());
171  return;
172  }
173  }
174  OMPL_WARN("No solution to simplify");
175 }

177 void ompl::geometric::SimpleSetup::simplifySolution(double duration)
178 {
179  if (pdef_)
180  {
181  const base::PathPtr &p = pdef_->getSolutionPath();
182  if (p)
183  {
184  time::point start = time::now();
185  PathGeometric &path = static_cast<PathGeometric&>(*p);
186  std::size_t numStates = path.getStateCount();
187  if (duration < std::numeric_limits<double>::epsilon())
188  psk_->simplifyMax(static_cast<PathGeometric&>(*p));
189  else
190  psk_->simplify(static_cast<PathGeometric&>(*p), duration);
191  simplifyTime_ = time::seconds(time::now() - start);
192  OMPL_INFORM("Path simplification took %f seconds and changed from %d to %d states",
193  simplifyTime_, numStates, path.getStateCount());
194  return;
195  }
196  }
197  OMPL_WARN("No solution to simplify");
198 }
