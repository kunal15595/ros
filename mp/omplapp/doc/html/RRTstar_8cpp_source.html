78 void ompl::geometric::RRTstar::setup()
79 {
80  Planner::setup();
81  tools::SelfConfig sc(si_, getName());
82  sc.configurePlannerRange(maxDistance_);
83 
84  if (!nn_)
85  nn_.reset(tools::SelfConfig::getDefaultNearestNeighbors<Motion*>(si_->getStateSpace()));
86  nn_->setDistanceFunction(boost::bind(&RRTstar::distanceFunction, this, _1, _2));
87 
88 
89  // Setup optimization objective
90  //
91  // If no optimization objective was specified, then default to
92  // optimizing path length as computed by the distance() function
93  // in the state space.
94  if(pdef_)
95  {
96  if (pdef_->hasOptimizationObjective())
97  opt_ = pdef_->getOptimizationObjective();
98  else
99  {
100  OMPL_INFORM("%s: No optimization objective specified. Defaulting to optimizing path length for the allowed planning time.", getName().c_str());
101  opt_.reset(new base::PathLengthOptimizationObjective(si_));
102  }
103  }
104  else
105  {
106  OMPL_INFORM("%s: problem definition is not set, deferring setup completion...", getName().c_str());
107  setup_ = false;
108  }
109 }

128 ompl::base::PlannerStatus ompl::geometric::RRTstar::solve(const base::PlannerTerminationCondition &ptc)
129 {
130  checkValidity();
131  base::Goal *goal = pdef_->getGoal().get();
132  base::GoalSampleableRegion *goal_s = dynamic_cast<base::GoalSampleableRegion*>(goal);
133 
134  bool symDist = si_->getStateSpace()->hasSymmetricDistance();
135  bool symInterp = si_->getStateSpace()->hasSymmetricInterpolate();
136  bool symCost = opt_->isSymmetric();
137 
138  while (const base::State *st = pis_.nextStart())
139  {
140  Motion *motion = new Motion(si_);
141  si_->copyState(motion->state, st);
142  motion->cost = opt_->identityCost();
143  nn_->add(motion);
144  }
145 
146  if (nn_->size() == 0)
147  {
148  OMPL_ERROR("%s: There are no valid initial states!", getName().c_str());
149  return base::PlannerStatus::INVALID_START;
150  }
151 
152  if (!sampler_)
153  sampler_ = si_->allocStateSampler();
154 
155  OMPL_INFORM("%s: Starting planning with %u states already in datastructure", getName().c_str(), nn_->size());
156 
157  Motion *solution = lastGoalMotion_;
158 
159  // \TODO Make this variable unnecessary, or at least have it
160  // persist across solve runs
161  base::Cost bestCost = opt_->infiniteCost();
162 
163  Motion *approximation = NULL;
164  double approximatedist = std::numeric_limits<double>::infinity();
165  bool sufficientlyShort = false;
166 
167  Motion *rmotion = new Motion(si_);
168  base::State *rstate = rmotion->state;
169  base::State *xstate = si_->allocState();
170 
171  // e+e/d. K-nearest RRT*
172  double k_rrg = boost::math::constants::e<double>() +
173  (boost::math::constants::e<double>() / (double)si_->getStateSpace()->getDimension());
174 
175  std::vector<Motion*> nbh;
176 
177  std::vector<base::Cost> costs;
178  std::vector<base::Cost> incCosts;
179  std::vector<std::size_t> sortedCostIndices;
180 
181  std::vector<int> valid;
182  unsigned int rewireTest = 0;
183  unsigned int statesGenerated = 0;
184 
185  if (solution)
186  OMPL_INFORM("%s: Starting planning with existing solution of cost %.5f", getName().c_str(), solution->cost.v);
187  OMPL_INFORM("%s: Initial k-nearest value of %u", getName().c_str(), (unsigned int)std::ceil(k_rrg * log((double)(nn_->size()+1))));
188 
189 
190  // our functor for sorting nearest neighbors
191  CostIndexCompare compareFn(costs, *opt_);
192 
193  while (ptc == false)
194  {
195  iterations_++;
196  // sample random state (with goal biasing)
197  // Goal samples are only sampled until maxSampleCount() goals are in the tree, to prohibit duplicate goal states.
198  if (goal_s && goalMotions_.size() < goal_s->maxSampleCount() && rng_.uniform01() < goalBias_ && goal_s->canSample())
199  goal_s->sampleGoal(rstate);
200  else
201  sampler_->sampleUniform(rstate);
202 
203  // Set directionality of nearest neighbors computation to be
204  // FROM neighbors TO new state
205  if (!symDist)
206  distanceDirection_ = FROM_NEIGHBORS;
207 
208  // find closest state in the tree
209  Motion *nmotion = nn_->nearest(rmotion);
210 
211  base::State *dstate = rstate;
212 
213  // find state to add to the tree
214  double d = si_->distance(nmotion->state, rstate);
215  if (d > maxDistance_)
216  {
217  si_->getStateSpace()->interpolate(nmotion->state, rstate, maxDistance_ / d, xstate);
218  dstate = xstate;
219  }
220 
221  // Check if the motion between the nearest state and the state to add is valid
222  ++collisionChecks_;
223  if (si_->checkMotion(nmotion->state, dstate))
224  {
225  // create a motion
226  Motion *motion = new Motion(si_);
227  si_->copyState(motion->state, dstate);
228  motion->parent = nmotion;
229  motion->incCost = opt_->motionCost(nmotion->state, motion->state);
230  motion->cost = opt_->combineCosts(nmotion->cost, motion->incCost);
231 
232  // Find nearby neighbors of the new motion - k-nearest RRT*
233  unsigned int k = std::ceil(k_rrg * log((double)(nn_->size() + 1)));
234  nn_->nearestK(motion, k, nbh);
235 
236  rewireTest += nbh.size();
237  statesGenerated++;
238 
239  // cache for distance computations
240  //
241  // Our cost caches only increase in size, so they're only
242  // resized if they can't fit the current neighborhood
243  if (costs.size() < nbh.size())
244  {
245  costs.resize(nbh.size());
246  incCosts.resize(nbh.size());
247  sortedCostIndices.resize(nbh.size());
248  }
249 
250  // cache for motion validity (only useful in a symmetric space)
251  //
252  // Our validity caches only increase in size, so they're
253  // only resized if they can't fit the current neighborhood
254  if (symDist && symInterp)
255  {
256  if (valid.size() < nbh.size())
257  valid.resize(nbh.size());
258  std::fill(valid.begin(), valid.begin()+nbh.size(), 0);
259  }
260 
261  // Finding the nearest neighbor to connect to
262  // By default, neighborhood states are sorted by cost, and collision checking
263  // is performed in increasing order of cost
264  if (delayCC_)
265  {
266  // calculate all costs and distances
267  for (std::size_t i = 0 ; i < nbh.size(); ++i)
268  {
269  incCosts[i] = opt_->motionCost(nbh[i]->state, motion->state);
270  costs[i] = opt_->combineCosts(nbh[i]->cost, incCosts[i]);
271  }
272 
273  // sort the nodes
274  //
275  // we're using index-value pairs so that we can get at
276  // original, unsorted indices
277  for (std::size_t i = 0; i < nbh.size(); ++i)
278  sortedCostIndices[i] = i;
279  std::sort(sortedCostIndices.begin(), sortedCostIndices.begin()+nbh.size(),
280  compareFn);
281 
282  // collision check until a valid motion is found
283  //
284  // ASYMMETRIC CASE: it's possible that none of these
285  // neighbors are valid. This is fine, because motion
286  // already has a connection to the tree through
287  // nmotion (with populated cost fields!).
288  for (std::vector<std::size_t>::const_iterator i = sortedCostIndices.begin();
289  i != sortedCostIndices.begin()+nbh.size();
290  ++i)
291  {
292  if (nbh[*i] != nmotion)
293  ++collisionChecks_;
294  if (nbh[*i] == nmotion || si_->checkMotion(nbh[*i]->state, motion->state))
295  {
296  motion->incCost = incCosts[*i];
297  motion->cost = costs[*i];
298  motion->parent = nbh[*i];
299  if (symDist && symInterp)
300  valid[*i] = 1;
301  break;
302  }
303  else if (symDist && symInterp)
304  valid[*i] = -1;
305  }
306  }
307  else // if not delayCC
308  {
309  motion->incCost = opt_->motionCost(nmotion->state, motion->state);
310  motion->cost = opt_->combineCosts(nmotion->cost, motion->incCost);
311  // find which one we connect the new state to
312  for (std::size_t i = 0 ; i < nbh.size(); ++i)
313  {
314  if (nbh[i] != nmotion)
315  {
316  incCosts[i] = opt_->motionCost(nbh[i]->state, motion->state);
317  costs[i] = opt_->combineCosts(nbh[i]->cost, incCosts[i]);
318  if (opt_->isCostBetterThan(costs[i], motion->cost))
319  {
320  ++collisionChecks_;
321  if (si_->checkMotion(nbh[i]->state, motion->state))
322  {
323  motion->incCost = incCosts[i];
324  motion->cost = costs[i];
325  motion->parent = nbh[i];
326  if (symDist && symInterp)
327  valid[i] = 1;
328  }
329  else if (symDist && symInterp)
330  valid[i] = -1;
331  }
332  }
333  else
334  {
335  incCosts[i] = motion->incCost;
336  costs[i] = motion->cost;
337  if (symDist && symInterp)
338  valid[i] = 1;
339  }
340  }
341  }
342 
343  // add motion to the tree
344  nn_->add(motion);
345  motion->parent->children.push_back(motion);
346 
347  bool checkForSolution = false;
348  // rewire tree if needed
349  //
350  // Set directionality of distance function to be FROM new
351  // state TO neighbors, since this is how the routing
352  // should occur in tree rewiring
353  if (!symDist)
354  {
355  distanceDirection_ = TO_NEIGHBORS;
356  nn_->nearestK(motion, k, nbh);
357  rewireTest += nbh.size();
358  }
359 
360  for (std::size_t i = 0; i < nbh.size(); ++i)
361  {
362  if (nbh[i] != motion->parent)
363  {
364  base::Cost nbhIncCost;
365  if (symDist && symCost)
366  nbhIncCost = incCosts[i];
367  else
368  nbhIncCost = opt_->motionCost(motion->state, nbh[i]->state);
369  base::Cost nbhNewCost = opt_->combineCosts(motion->cost, nbhIncCost);
370  if (opt_->isCostBetterThan(nbhNewCost, nbh[i]->cost))
371  {
372  bool motionValid;
373  if (symDist && symInterp)
374  {
375  if (valid[i] == 0)
376  {
377  ++collisionChecks_;
378  motionValid = si_->checkMotion(motion->state, nbh[i]->state);
379  }
380  else
381  motionValid = (valid[i] == 1);
382 
383  }
384  else
385  {
386  ++collisionChecks_;
387  motionValid = si_->checkMotion(motion->state, nbh[i]->state);
388  }
389  if (motionValid)
390  {
391  // Remove this node from its parent list
392  removeFromParent (nbh[i]);
393 
394  // Add this node to the new parent
395  nbh[i]->parent = motion;
396  nbh[i]->incCost = nbhIncCost;
397  nbh[i]->cost = nbhNewCost;
398  nbh[i]->parent->children.push_back(nbh[i]);
399 
400  // Update the costs of the node's children
401  updateChildCosts(nbh[i]);
402 
403  checkForSolution = true;
404  }
405  }
406  }
407  }
408 
409  // Add the new motion to the goalMotion_ list, if it satisfies the goal
410  double distanceFromGoal;
411  if (goal->isSatisfied(motion->state, &distanceFromGoal))
412  {
413  goalMotions_.push_back(motion);
414  checkForSolution = true;
415  }
416 
417  // Checking for solution or iterative improvement
418  if (checkForSolution)
419  {
420  for (size_t i = 0; i < goalMotions_.size(); ++i)
421  {
422  if (opt_->isCostBetterThan(goalMotions_[i]->cost, bestCost))
423  {
424  bestCost = goalMotions_[i]->cost;
425  bestCost_ = bestCost;
426  }
427 
428  sufficientlyShort = opt_->isSatisfied(goalMotions_[i]->cost);
429  if (sufficientlyShort)
430  {
431  solution = goalMotions_[i];
432  break;
433  }
434  else if (!solution ||
435  opt_->isCostBetterThan(goalMotions_[i]->cost,solution->cost))
436  solution = goalMotions_[i];
437  }
438  }
439 
440  // Checking for approximate solution (closest state found to the goal)
441  if (goalMotions_.size() == 0 && distanceFromGoal < approximatedist)
442  {
443  approximation = motion;
444  approximatedist = distanceFromGoal;
445  }
446  }
447 
448  // terminate if a sufficient solution is found
449  if (solution && sufficientlyShort)
450  break;
451  }
452 
453  bool approximate = (solution == 0);
454  bool addedSolution = false;
455  if (approximate)
456  solution = approximation;
457  else
458  lastGoalMotion_ = solution;
459 
460  if (solution != 0)
461  {
462  // construct the solution path
463  std::vector<Motion*> mpath;
464  while (solution != 0)
465  {
466  mpath.push_back(solution);
467  solution = solution->parent;
468  }
469 
470  // set the solution path
471  PathGeometric *geoPath = new PathGeometric(si_);
472  for (int i = mpath.size() - 1 ; i >= 0 ; --i)
473  geoPath->append(mpath[i]->state);
474 
475  base::PathPtr path(geoPath);
476  // Add the solution path, whether it is approximate (not reaching the goal), and the
477  // distance from the end of the path to the goal (-1 if satisfying the goal).
478  base::PlannerSolution psol(path, approximate, approximate ? approximatedist : -1.0, getName());
479  // Does the solution satisfy the optimization objective?
480  psol.optimized_ = sufficientlyShort;
481 
482  pdef_->addSolutionPath (psol);
483 
484  addedSolution = true;
485  }
486 
487  si_->freeState(xstate);
488  if (rmotion->state)
489  si_->freeState(rmotion->state);
490  delete rmotion;
491 
492  OMPL_INFORM("%s: Created %u new states. Checked %u rewire options. %u goal states in tree.", getName().c_str(), statesGenerated, rewireTest, goalMotions_.size());
493 
494  return base::PlannerStatus(addedSolution, approximate);
495 }

498 {
499  std::vector<Motion*>::iterator it = m->parent->children.begin ();
500  while (it != m->parent->children.end ())
501  {
502  if (*it == m)
503  {
504  it = m->parent->children.erase(it);
505  it = m->parent->children.end ();
506  }
507  else
508  ++it;
