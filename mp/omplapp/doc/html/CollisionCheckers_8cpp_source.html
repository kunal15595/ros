139 void test (unsigned int tries, std::vector <std::vector<double> > &times, std::vector<int> &attempts, bool fcl=false, bool continuous=false)
140 {
141  std::cout << "Evaluating " << (continuous ? "continuous" : "discrete")
142  << " " << (fcl ? "FCL" : "PQP") << " checker" << std::endl;
143 
144  std::vector <double> time;
145  unsigned int successful = 0;
146  unsigned int problem = 0;
147 
148  while (problem != 3)
149  {
150  unsigned int nr_attempts = 0;
151 
152  if (problem == 0) std::cout << "- Apartment problem " << std::flush;
153  else if (problem == 1) std::cout << "- Cubicles problem " << std::flush;
154  else if (problem == 2) std::cout << "- \'Easy\' problem " << std::flush;
155 
156  // plan in SE3
157  app::SE3RigidBodyPlanning setup;
158 
159  switch (problem)
160  {
161  case 0: configureApartmentProblem (setup); break;
162  case 1: configureCubiclesProblem (setup); break;
163  case 2: configureEasyProblem (setup); break;
164  }
165 
166  setup.setStateValidityCheckerType (fcl ? app::FCL : app::PQP);
167 
168  if (continuous)
169  {
170  setup.setup(); // FCLContinuousMotionValidator extracts goodies from the state validity checker. Instantiate the svc here.
171  base::MotionValidatorPtr mv (new app::FCLContinuousMotionValidator (setup.getSpaceInformation(), setup.getMotionModel()));
172  setup.getSpaceInformation ()->setMotionValidator (mv);
173  }
174 
175  setup.getSpaceInformation()->setStateValidityCheckingResolution(0.01);
176  setup.setPlanner (base::PlannerPtr (new geometric::RRT (setup.getSpaceInformation ())));
177  setup.setup ();
178 
179  while (successful < tries)
180  {
181  setup.clear ();
182  setup.solve(30.0);
183 
184  // Retry if the planner failed, except for the continuous collision checker case.
185  if (setup.haveExactSolutionPath () || continuous)
186  {
187  successful++;
188  time.push_back (setup.getLastPlanComputationTime ());
189  std::cout << '.' << std::flush;
190  }
191  else
192  std::cout << 'x' << std::flush;
193  nr_attempts++;
194  }
195 
196  times.push_back (time);
197  time.clear ();
198  successful = 0;
199  problem++;
200  attempts.push_back (nr_attempts);
201  std::cout << std::endl;
202  }
203 }

205 // Compares discrete and continuous collision checkers in OMPL.app
206 int main (int argc, char **argv)
207 {
208  // User can supply number of tries as 2nd command line argument. Otherwise, use default NR_TRIES.
209  unsigned int nr_tries;
210  if (argc == 2)
211  {
212  nr_tries = atoi (argv[1]);
213 
214  // make sure command line input is valid
215  if (nr_tries == std::numeric_limits<unsigned int>::min() || nr_tries == std::numeric_limits<unsigned int>::max())
216  nr_tries = 20u;
217  }
218  else
219  nr_tries = 20u;
220 
221  ompl::msg::noOutputHandler(); // Disable console output from OMPL.
222 
223  std::vector <std::vector <double> > pqp_times;
224  std::vector <std::vector <double> > dfcl_times;
225  std::vector <std::vector <double> > cfcl_times;
226  std::vector <int> pqp_attempts;
227  std::vector <int> dfcl_attempts;
228  std::vector <int> cfcl_attempts;
229 
230  std::cout << "Comparing collision checkers:" << std::endl;
231  std::cout << "Each problem is executed until " << nr_tries << " attempts are successful (30 sec limit)" << std::endl;
232 
233  // PQP Test (discrete)
234  test (nr_tries, pqp_times, pqp_attempts);
235 
236  // Discrete FCL Test
237  test (nr_tries, dfcl_times, dfcl_attempts, true);
238 
239  // Continuous FCL Test
240  test (nr_tries, cfcl_times, cfcl_attempts, true, true);
241 
242  std::cout << std::endl << "Analysis:" << std::endl;
243 
244  // Assume size of all 3 time vectors are the same
245  for (size_t i = 0; i < pqp_times.size (); ++i)
246  {
247  bool isOdd = pqp_times[i].size() % 2 == 1;
248  unsigned int n = pqp_times[i].size() / 2u;
249  std::vector<double>::iterator pqp_time = pqp_times[i].begin() + n;
250  std::vector<double>::iterator dfcl_time = dfcl_times[i].begin()+n;
251  std::vector<double>::iterator cfcl_time = cfcl_times[i].begin()+n;
252 
253  std::nth_element(pqp_times[i].begin(), pqp_time, pqp_times[i].end());
254  std::nth_element(dfcl_times[i].begin(), dfcl_time, dfcl_times[i].end());
255  std::nth_element(cfcl_times[i].begin(), cfcl_time, cfcl_times[i].end());
256 
257  std::cout << std::endl;
258  if (i == 0)
259  std::cout << " Apartment problem - Median Time (s)" << std::endl;
260  else if (i == 1)
261  std::cout << " Cubicles problem - Median Time (s)" << std::endl;
262  else if (i == 2)
263  std::cout << " \'Easy\' problem - Median Time (s)" << std::endl;
264 
265  std::cout << " Discrete PQP: " << (isOdd ? *pqp_time : .5*(*pqp_time + *(pqp_time+1)))
266  << " " << nr_tries << "/" << pqp_attempts[i] << " planning attempts successful" << std::endl;
267  std::cout << " Discrete FCL: " << (isOdd ? *dfcl_time : .5*(*dfcl_time + *(dfcl_time+1)))
268  << " " << nr_tries << "/" << dfcl_attempts[i] << " planning attempts successful" << std::endl;
269  std::cout << " Continuous FCL: " << (isOdd ? *cfcl_time : .5*(*cfcl_time + *(cfcl_time+1)))
270  << " " << nr_tries << " total attempts" << std::endl;
271  }
272 
273  return 0;
274 }
