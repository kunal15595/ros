53 void benchmark0(std::string& benchmark_name, app::SE3RigidBodyPlanning& setup,
54  double& runtime_limit, double& memory_limit, int& run_count)
55 {
56  benchmark_name = std::string("cubicles");
57  std::string robot_fname = std::string(OMPLAPP_RESOURCE_DIR) + "/3D/cubicles_robot.dae";
58  std::string env_fname = std::string(OMPLAPP_RESOURCE_DIR) + "/3D/cubicles_env.dae";
59  setup.setRobotMesh(robot_fname.c_str());
60  setup.setEnvironmentMesh(env_fname.c_str());
61 
62  base::ScopedState<base::SE3StateSpace> start(setup.getSpaceInformation());
63  start->setX(-4.96);
64  start->setY(-40.62);
65  start->setZ(70.57);
66  start->rotation().setIdentity();
67 
68  base::ScopedState<base::SE3StateSpace> goal(start);
69  goal->setX(200.49);
70  goal->setY(-40.62);
71  goal->setZ(70.57);
72  goal->rotation().setIdentity();
73 
74  setup.setStartAndGoalStates(start, goal);
75  setup.getSpaceInformation()->setStateValidityCheckingResolution(0.01);
76  setup.getSpaceInformation()->setValidStateSamplerAllocator(&allocUniformStateSampler);
77  setup.setup();
78 
79  std::vector<double> cs(3);
80  cs[0] = 35; cs[1] = 35; cs[2] = 35;
81  setup.getStateSpace()->getDefaultProjection()->setCellSizes(cs);
82 
83  runtime_limit = 10.0;
84  memory_limit = 10000.0; // set high because memory usage is not always estimated correctly
85  run_count = 500;
86 }

134 int main(int argc, char **argv)
135 {
136  app::SE3RigidBodyPlanning setup;
137  std::string benchmark_name;
138  double runtime_limit, memory_limit;
139  int run_count;
140  int benchmark_id = argc > 1 ? ((argv[1][0] - '0') % 2) : 0;
141 
142  if (benchmark_id==0)
143  benchmark0(benchmark_name, setup, runtime_limit, memory_limit, run_count);
144  else
145  benchmark1(benchmark_name, setup, runtime_limit, memory_limit, run_count);
146 
147  // create the benchmark object and add all the planners we'd like to run
148  tools::Benchmark::Request request(runtime_limit, memory_limit, run_count);
149  tools::Benchmark b(setup, benchmark_name);
150 
151  // optionally set pre & pos run events
152  b.setPreRunEvent(boost::bind(&preRunEvent, _1));
153  b.setPostRunEvent(boost::bind(&postRunEvent, _1, _2));
154 
155  b.addPlanner(base::PlannerPtr(new geometric::RRTConnect(setup.getSpaceInformation())));
156  b.addPlanner(base::PlannerPtr(new geometric::RRT(setup.getSpaceInformation())));
157  b.addPlanner(base::PlannerPtr(new geometric::BKPIECE1(setup.getSpaceInformation())));
158  b.addPlanner(base::PlannerPtr(new geometric::LBKPIECE1(setup.getSpaceInformation())));
159  b.addPlanner(base::PlannerPtr(new geometric::KPIECE1(setup.getSpaceInformation())));
160  b.addPlanner(base::PlannerPtr(new geometric::SBL(setup.getSpaceInformation())));
161  b.addPlanner(base::PlannerPtr(new geometric::EST(setup.getSpaceInformation())));
162  b.addPlanner(base::PlannerPtr(new geometric::PRM(setup.getSpaceInformation())));
163 
164  int sampler_id = argc > 2 ? ((argv[2][0] - '0') % 4) : -1;
165 
166  if (sampler_id == 0 || sampler_id < 0)
167  {
168  // run all planners with a uniform valid state sampler on the benchmark problem
169  setup.getSpaceInformation()->setValidStateSamplerAllocator(&allocUniformStateSampler);
170  b.setExperimentName(benchmark_name + "_uniform_sampler");
171  b.benchmark(request);
172  b.saveResultsToFile();
173  }
174 
175  if (sampler_id == 1 || sampler_id < 0)
176  {
177  // run all planners with a Gaussian valid state sampler on the benchmark problem
178  setup.getSpaceInformation()->setValidStateSamplerAllocator(&allocGaussianStateSampler);
179  b.setExperimentName(benchmark_name + "_gaussian_sampler");
180  b.benchmark(request);
181  b.saveResultsToFile();
182  }
183 
184  if (sampler_id == 2 || sampler_id < 0)
185  {
186  // run all planners with a obstacle-based valid state sampler on the benchmark problem
187  setup.getSpaceInformation()->setValidStateSamplerAllocator(&allocObstacleStateSampler);
188  b.setExperimentName(benchmark_name + "_obstaclebased_sampler");
189  b.benchmark(request);
190  b.saveResultsToFile();
191  }
192 
193  if (sampler_id == 3 || sampler_id < 0)
194  {
195  // run all planners with a maximum-clearance valid state sampler on the benchmark problem
196  setup.getSpaceInformation()->setValidStateSamplerAllocator(&allocMaximizeClearanceStateSampler);
197  b.setExperimentName(benchmark_name + "_maxclearance_sampler");
198  b.benchmark(request);
199  b.saveResultsToFile();
200  }
201 
202  return 0;
203 }
