184 bool ompl::base::ProblemDefinition::fixInvalidInputState(State *state, double dist, bool start, unsigned int attempts)
185 {
186  bool result = false;
187 
188  bool b = si_->satisfiesBounds(state);
189  bool v = false;
190  if (b)
191  {
192  v = si_->isValid(state);
193  if (!v)
194  OMPL_DEBUG("%s state is not valid", start ? "Start" : "Goal");
195  }
196  else
197  OMPL_DEBUG("%s state is not within space bounds", start ? "Start" : "Goal");
198 
199  if (!b || !v)
200  {
201  std::stringstream ss;
202  si_->printState(state, ss);
203  ss << " within distance " << dist;
204  OMPL_DEBUG("Attempting to fix %s state %s", start ? "start" : "goal", ss.str().c_str());
205 
206  State *temp = si_->allocState();
207  if (si_->searchValidNearby(temp, state, dist, attempts))
208  {
209  si_->copyState(state, temp);
210  result = true;
211  }
212  else
213  OMPL_WARN("Unable to fix %s state", start ? "start" : "goal");
214  si_->freeState(temp);
215  }
216 
217  return result;
218 }

220 bool ompl::base::ProblemDefinition::fixInvalidInputStates(double distStart, double distGoal, unsigned int attempts)
221 {
222  bool result = true;
223 
224  // fix start states
225  for (unsigned int i = 0 ; i < startStates_.size() ; ++i)
226  if (!fixInvalidInputState(startStates_[i], distStart, true, attempts))
227  result = false;
228 
229  // fix goal state
230  GoalState *goal = dynamic_cast<GoalState*>(goal_.get());
231  if (goal)
232  {
233  if (!fixInvalidInputState(const_cast<State*>(goal->getState()), distGoal, false, attempts))
234  result = false;
235  }
236 
237  // fix goal state
238  GoalStates *goals = dynamic_cast<GoalStates*>(goal_.get());
239  if (goals)
240  {
241  for (unsigned int i = 0; i < goals->getStateCount(); ++i)
242  if (!fixInvalidInputState(const_cast<State*>(goals->getState(i)), distGoal, false, attempts))
243  result = false;
244  }
245 
246  return result;
247 }

265 ompl::base::PathPtr ompl::base::ProblemDefinition::isStraightLinePathValid() const
266 {
267  PathPtr path;
268  if (control::SpaceInformationPtr sic = boost::dynamic_pointer_cast<control::SpaceInformation, SpaceInformation>(si_))
269  {
270  unsigned int startIndex;
271  if (isTrivial(&startIndex, NULL))
272  {
273  control::PathControl *pc = new control::PathControl(sic);
274  pc->append(startStates_[startIndex]);
275  control::Control *null = sic->allocControl();
276  sic->nullControl(null);
277  pc->append(startStates_[startIndex], null, 0.0);
278  sic->freeControl(null);
279  path.reset(pc);
280  }
281  else
282  {
283  control::Control *nc = sic->allocControl();
284  State *result1 = sic->allocState();
285  State *result2 = sic->allocState();
286  sic->nullControl(nc);
287 
288  for (unsigned int k = 0 ; k < startStates_.size() && !path ; ++k)
289  {
290  const State *start = startStates_[k];
291  if (start && si_->isValid(start) && si_->satisfiesBounds(start))
292  {
293  sic->copyState(result1, start);
294  for (unsigned int i = 0 ; i < sic->getMaxControlDuration() && !path ; ++i)
295  if (sic->propagateWhileValid(result1, nc, 1, result2))
296  {
297  if (goal_->isSatisfied(result2))
298  {
299  control::PathControl *pc = new control::PathControl(sic);
300  pc->append(start);
301  pc->append(result2, nc, (i + 1) * sic->getPropagationStepSize());
302  path.reset(pc);
303  break;
304  }
305  std::swap(result1, result2);
306  }
307  }
308  }
309  sic->freeState(result1);
310  sic->freeState(result2);
311  sic->freeControl(nc);
312  }
313  }
314  else
315  {
316  std::vector<const State*> states;
317  GoalState *goal = dynamic_cast<GoalState*>(goal_.get());
318  if (goal)
319  if (si_->isValid(goal->getState()) && si_->satisfiesBounds(goal->getState()))
320  states.push_back(goal->getState());
321  GoalStates *goals = dynamic_cast<GoalStates*>(goal_.get());
322  if (goals)
323  for (unsigned int i = 0; i < goals->getStateCount(); ++i)
324  if (si_->isValid(goals->getState(i)) && si_->satisfiesBounds(goals->getState(i)))
325  states.push_back(goals->getState(i));
326 
327  if (states.empty())
328  {
329  unsigned int startIndex;
330  if (isTrivial(&startIndex))
331  {
332  geometric::PathGeometric *pg = new geometric::PathGeometric(si_, startStates_[startIndex], startStates_[startIndex]);
333  path.reset(pg);
334  }
335  }
336  else
337  {
338  for (unsigned int i = 0 ; i < startStates_.size() && !path ; ++i)
339  {
340  const State *start = startStates_[i];
341  if (start && si_->isValid(start) && si_->satisfiesBounds(start))
342  {
343  for (unsigned int j = 0 ; j < states.size() && !path ; ++j)
344  if (si_->checkMotion(start, states[j]))
345  {
346  geometric::PathGeometric *pg = new geometric::PathGeometric(si_, start, states[j]);
347  path.reset(pg);
348  break;
349  }
350  }
351  }
352  }
353  }
354 
355  return path;
356 }

358 bool ompl::base::ProblemDefinition::isTrivial(unsigned int *startIndex, double *distance) const
359 {
360  if (!goal_)
361  {
362  OMPL_ERROR("Goal undefined");
363  return false;
364  }
365 
366  for (unsigned int i = 0 ; i < startStates_.size() ; ++i)
367  {
368  const State *start = startStates_[i];
369  if (start && si_->isValid(start) && si_->satisfiesBounds(start))
370  {
371  double dist;
372  if (goal_->isSatisfied(start, &dist))
373  {
374  if (startIndex)
375  *startIndex = i;
376  if (distance)
377  *distance = dist;
378  return true;
379  }
380  }
381  else
382  {
383  OMPL_ERROR("Initial state is in collision!");
384  }
385  }
386 
387  return false;
388 }
