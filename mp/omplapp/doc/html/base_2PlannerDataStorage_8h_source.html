177  virtual void loadVertices(PlannerData &pd, unsigned int numVertices, boost::archive::binary_iarchive &ia)
178  {
179  const StateSpacePtr &space = pd.getSpaceInformation()->getStateSpace();
180  std::vector<State*> states;
181  for (unsigned int i = 0; i < numVertices; ++i)
182  {
183  PlannerDataVertexData vertexData;
184  ia >> vertexData;
185 
186  // Deserializing all data in the vertex (except the state)
187  const PlannerDataVertex *v = vertexData.v_;
188 
189  // Allocating a new state and deserializing it from the buffer
190  State *state = space->allocState();
191  states.push_back(state);
192  space->deserialize (state, &vertexData.state_[0]);
193  const_cast<PlannerDataVertex*>(v)->state_ = state;
194 
195  // Record the type of the vertex (i.e. start vertex).
196  if (vertexData.type_ == PlannerDataVertexData::START)
197  pd.addStartVertex(*v);
198  else if (vertexData.type_ == PlannerDataVertexData::GOAL)
199  pd.addGoalVertex(*v);
200  else
201  pd.addVertex(*v);
202 
203  // We deserialized the vertex object pointer, and we own it.
204  // Since addEdge copies the object, it is safe to free here.
205  delete vertexData.v_;
206  }
207 
208  // These vertices are using state pointers allocated here.
209  // To avoid a memory leak, we decouple planner data from the
210  // 'planner', which will clone all states and properly free the
211  // memory when PlannerData goes out of scope. Then it is safe
212  // to free all memory allocated here.
213  pd.decoupleFromPlanner();
214 
215  for (size_t i = 0; i < states.size(); ++i)
216  space->freeState(states[i]);
217  }

220  virtual void storeVertices(const PlannerData &pd, boost::archive::binary_oarchive &oa)
221  {
222  const StateSpacePtr &space = pd.getSpaceInformation()->getStateSpace();
223  std::vector<unsigned char> state (space->getSerializationLength());
224  for (unsigned int i = 0; i < pd.numVertices(); ++i)
225  {
226  PlannerDataVertexData vertexData;
227 
228  // Serializing all data in the vertex (except the state)
229  const PlannerDataVertex &v = pd.getVertex(i);
230  vertexData.v_ = &v;
231 
232  // Record the type of the vertex (i.e. start vertex).
233  if (pd.isStartVertex(i))
234  vertexData.type_ = PlannerDataVertexData::START;
235  else if (pd.isGoalVertex(i))
236  vertexData.type_ = PlannerDataVertexData::GOAL;
237  else vertexData.type_ = PlannerDataVertexData::STANDARD;
238 
239  // Serializing the state contained in this vertex
240  space->serialize (&state[0], v.getState());
241  vertexData.state_ = state;
242 
243  oa << vertexData;
244  }
245  }

248  virtual void loadEdges(PlannerData &pd, unsigned int numEdges, boost::archive::binary_iarchive &ia)
249  {
250  for (unsigned int i = 0; i < numEdges; ++i)
251  {
252  PlannerDataEdgeData edgeData;
253  ia >> edgeData;
254  pd.addEdge(edgeData.endpoints_.first, edgeData.endpoints_.second, *edgeData.e_, Cost(edgeData.weight_));
255 
256  // We deserialized the edge object pointer, and we own it.
257  // Since addEdge copies the object, it is safe to free here.
258  delete edgeData.e_;
259  }
260  }

263  virtual void storeEdges(const PlannerData &pd, boost::archive::binary_oarchive &oa)
264  {
265  for (unsigned int i = 0; i < pd.numVertices(); ++i)
266  for (unsigned int j = 0; j < pd.numVertices(); ++j)
267  {
268  if(pd.edgeExists(i, j))
269  {
270  PlannerDataEdgeData edgeData;
271  edgeData.e_ = &pd.getEdge(i, j);
272  edgeData.endpoints_.first = i;
273  edgeData.endpoints_.second = j;
274  Cost weight;
275  pd.getEdgeWeight(i, j, &weight);
276  edgeData.weight_ = weight.v;
277 
278  oa << edgeData;
279  }
280  }
281  }
282  };
